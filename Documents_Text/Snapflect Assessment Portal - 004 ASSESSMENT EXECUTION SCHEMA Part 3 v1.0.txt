QUESTION_RESPONSES
Purpose
Stores the latest candidate response for a runtime question.
Response history is intentionally not stored.
Only the most recent response is maintained.
Response changes are tracked through:
assessment_attempt_questions.response_count

CREATETABLE question_responses(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    attempt_question_id BIGINT UNSIGNED NOTNULL,    response_value LONGTEXT NULL,    response_status VARCHAR(50)NOTNULLDEFAULT'Draft',    response_date DATETIME(3) NULL,    evaluated_date DATETIME(3) NULL,    requires_manual_review TINYINT(1)NOTNULLDEFAULT0,    is_final_answer TINYINT(1)NOTNULLDEFAULT0,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),    modified_date DATETIME(3)NULL,PRIMARYKEY (id),CONSTRAINT ux_question_responses_uuidUNIQUE (uuid),CONSTRAINT ux_question_responses_attempt_questionUNIQUE (attempt_question_id))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

DRAFT_SAVES
Purpose
Stores the latest recoverable assessment draft.
Supports:
Auto SaveManual SaveResumeRecoveryFuture Mobile Offline Sync

Storage Strategy
One active draft per assessment attempt.
New saves overwrite previous snapshots.

CREATETABLE draft_saves(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    attempt_id BIGINT UNSIGNED NOTNULL,    draft_version INTNOTNULLDEFAULT1,    client_save_guid CHAR(36) NULL,    payload_checksum VARCHAR(128) NULL,    last_saved_question_id BIGINT UNSIGNED NULL,    save_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),    draft_payload_json JSON NOTNULL,PRIMARYKEY (id),CONSTRAINT ux_draft_saves_uuidUNIQUE (uuid),CONSTRAINT ux_draft_saves_attemptUNIQUE (attempt_id),CONSTRAINT ux_draft_saves_client_guidUNIQUE (client_save_guid))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

ALTERTABLE draft_savesADDCONSTRAINT fk_draft_saves_attemptFOREIGNKEY (attempt_id)REFERENCES assessment_attempts(id);ALTERTABLE draft_savesADDCONSTRAINT fk_draft_saves_last_questionFOREIGNKEY (last_saved_question_id)REFERENCES assessment_attempt_questions(id);

DRAFT_SAVES INDEXES
