INVOICES
Purpose
Stores immutable customer invoices.

Governance
Invoices are financial records.
Issued invoices must never be modified directly.
Corrections should be handled through adjustments, credits, or replacement invoices.

CREATETABLE invoices(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    correlation_id CHAR(36) NULL,    customer_subscription_id BIGINT UNSIGNED NOTNULL,    invoice_number VARCHAR(100) NOTNULL,    invoice_date DATETIME(3) NOTNULL,    billing_period_start DATETIME(3) NOTNULL,    billing_period_end DATETIME(3) NOTNULL,    subtotal_amount DECIMAL(18,2)NOTNULLDEFAULT0.00,    tax_percentage DECIMAL(8,4)NOTNULLDEFAULT0.0000,    tax_amount DECIMAL(18,2)NOTNULLDEFAULT0.00,    total_amount DECIMAL(18,2)NOTNULLDEFAULT0.00,    currency_code CHAR(3)NOTNULLDEFAULT'INR',    subscription_snapshot_json JSON NOTNULL,    status VARCHAR(50)NOTNULLDEFAULT'Draft',    metadata_json JSON NULL,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),PRIMARYKEY (id),CONSTRAINT ux_invoices_uuidUNIQUE (uuid),CONSTRAINT ux_invoices_invoice_numberUNIQUE (invoice_number))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

PAYMENTS
Purpose
Stores customer payment records.

Governance
A payment may be associated with one invoice.
Provider-specific details belong in payment_transactions.

CREATETABLE payments(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    correlation_id CHAR(36) NULL,    invoice_id BIGINT UNSIGNED NOTNULL,    payment_amount DECIMAL(18,2)NOTNULL,    payment_method VARCHAR(50) NOTNULL,    payment_date DATETIME(3) NOTNULL,    refund_amount DECIMAL(18,2)NOTNULLDEFAULT0.00,    refund_date DATETIME(3) NULL,    status VARCHAR(50)NOTNULLDEFAULT'Pending',    metadata_json JSON NULL,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),PRIMARYKEY (id),CONSTRAINT ux_payments_uuidUNIQUE (uuid))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

PAYMENT_TRANSACTIONS
Purpose
Stores payment gateway transaction details.

Governance
Provider transactions are immutable.
Used for reconciliation and dispute resolution.

CREATETABLE payment_transactions(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    correlation_id CHAR(36) NULL,    payment_id BIGINT UNSIGNED NOTNULL,    provider_name VARCHAR(100) NOTNULL,    provider_transaction_reference VARCHAR(255) NOTNULL,    transaction_status VARCHAR(50) NOTNULL,    reconciliation_status VARCHAR(50)NOTNULLDEFAULT'Pending',    transaction_amount DECIMAL(18,2)NOTNULL,    transaction_date DATETIME(3) NOTNULL,    provider_response_json JSON NULL,    metadata_json JSON NULL,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),PRIMARYKEY (id),CONSTRAINT ux_payment_transactions_uuidUNIQUE (uuid),CONSTRAINT ux_payment_transactions_referenceUNIQUE        (            provider_name,            provider_transaction_reference        ))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;
