QUESTIONS
Purpose
Represents reusable question master entities.
Questions are versioned through:
question_versions
Question content must never be stored directly in the master table.

CREATETABLE questions(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    question_code VARCHAR(100) NOTNULL,    status VARCHAR(50)NOTNULLDEFAULT'Draft',    created_by BIGINT UNSIGNED NOTNULL,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),    modified_by BIGINT UNSIGNED NULL,    modified_date DATETIME(3) NULL,    is_active TINYINT(1)NOTNULLDEFAULT1,    is_deleted TINYINT(1)NOTNULLDEFAULT0,    deleted_by BIGINT UNSIGNED NULL,    deleted_date DATETIME(3) NULL,PRIMARYKEY (id),CONSTRAINT ux_questions_uuidUNIQUE (uuid),CONSTRAINT ux_questions_codeUNIQUE (question_code))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

QUESTION_VERSIONS
Purpose
Stores complete historical snapshots of question definitions.
Uses Full Snapshot Versioning Architecture.

CREATETABLE question_versions(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    question_id BIGINT UNSIGNED NOTNULL,    question_bank_version_id BIGINT UNSIGNED NOTNULL,    version_number INTNOTNULL,    version_status VARCHAR(50)NOTNULLDEFAULT'Draft',    question_text LONGTEXT NOTNULL,    question_type VARCHAR(50) NOTNULL,    difficulty_level VARCHAR(50)NOTNULLDEFAULT'Medium',    marks DECIMAL(10,2)NOTNULLDEFAULT1.00,    negative_marks DECIMAL(10,2)NOTNULLDEFAULT0.00,    instructions TEXT NULL,    estimated_duration_minutes INTNULL,    display_order INTNOTNULLDEFAULT0,    is_randomizable TINYINT(1)NOTNULLDEFAULT1,    metadata_json JSON NULL,    created_by BIGINT UNSIGNED NOTNULL,    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),PRIMARYKEY (id),CONSTRAINT ux_question_versions_uuidUNIQUE (uuid),CONSTRAINT ux_question_versions_versionUNIQUE (question_id, version_number))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;

ALTERTABLE question_versionsADDCONSTRAINT fk_question_versions_questionFOREIGNKEY (question_id)REFERENCES questions(id);ALTERTABLE question_versionsADDCONSTRAINT fk_question_versions_question_bank_versionFOREIGNKEY (question_bank_version_id)REFERENCES question_bank_versions(id);ALTERTABLE question_versionsADDCONSTRAINT fk_question_versions_created_byFOREIGNKEY (created_by)REFERENCES users(id);

QUESTION_OPTIONS
Purpose
Stores selectable answer options.
Options belong to a Question Version.
This preserves historical version integrity.

CREATETABLE question_options(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    question_version_id BIGINT UNSIGNED NOTNULL,    option_code VARCHAR(20) NOTNULL,    option_text LONGTEXT NOTNULL,    display_order INTNOTNULLDEFAULT0,    is_active TINYINT(1)NOTNULLDEFAULT1,PRIMARYKEY (id),CONSTRAINT ux_question_options_uuidUNIQUE (uuid),CONSTRAINT ux_question_option_codeUNIQUE (question_version_id, option_code))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;
