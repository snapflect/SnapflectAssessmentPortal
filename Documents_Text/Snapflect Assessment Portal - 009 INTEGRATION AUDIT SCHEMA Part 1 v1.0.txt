API_KEYS
Purpose
Stores hashed API credentials.
Supports:
Key RotationKey ExpiryKey RevocationScope-Based Authorization

Security Rule
Plaintext API keys must never be stored.
Only hashed values may be persisted.

CREATETABLE api_keys(id BIGINT UNSIGNED NOTNULL AUTO_INCREMENT,    uuid CHAR(36) NOTNULL,    api_client_id BIGINT UNSIGNED NOTNULL,    key_hash VARCHAR(512) NOTNULL,    scope_json JSON NOTNULL,    expiry_date DATETIME(3) NULL,    rotated_from_key_id BIGINT UNSIGNED NULL,    status VARCHAR(50)NOTNULLDEFAULT'Active',    created_date DATETIME(3)NOTNULLDEFAULTCURRENT_TIMESTAMP(3),    revoked_date DATETIME(3) NULL,PRIMARYKEY (id),CONSTRAINT ux_api_keys_uuidUNIQUE (uuid),CONSTRAINT ux_api_keys_hashUNIQUE (key_hash))ENGINE=InnoDBDEFAULT CHARSET=utf8mb4COLLATE=utf8mb4_unicode_ci;
